func (r RetryableHTTPStatusCode) IsErrorRetryable(err error) aws.Ternary {
	var v interface{ HTTPStatusCode() int }

	if !errors.As(err, &v) {
		return aws.UnknownTernary
	}

	_, ok := r.Codes[v.HTTPStatusCode()]
	if !ok {
		return aws.UnknownTernary
	}

	return aws.TrueTernary
}
