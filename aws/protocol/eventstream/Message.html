func (m Message) Clone() Message {
	var payload []byte
	if m.Payload != nil {
		payload = make([]byte, len(m.Payload))
		copy(payload, m.Payload)
	}

	return Message{
		Headers: m.Headers.Clone(),
		Payload: payload,
	}
}
